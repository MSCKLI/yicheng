        let selectedStudent = null;

        function selectStudent(student) {
            selectedStudent = student;
            
            // 更新选中状态
            document.querySelectorAll('.pet-card').forEach(c => c.classList.remove('selected'));
            event.currentTarget.classList.add('selected');
            
            // 更新底部控制栏
            document.getElementById('selectedPetAvatar').textContent = student.pet.type.emoji;
            document.getElementById('selectedPetName').textContent = `${student.name}的${student.pet.name}`;
            document.getElementById('selectedPetLevel').textContent = `Lv.${student.pet.level}`;
            
            const stageName = student.pet.type.stages[Math.min(student.pet.stage, student.pet.type.stages.length - 1)];
            document.getElementById('selectedPetStage').textContent = stageName;
            
            // 启用按钮
            ['btnFeed', 'btnPlay', 'btnStudy'].forEach(id => {
                document.getElementById(id).disabled = false;
                document.getElementById(id).classList.remove('opacity-50');
            });
            
            // 检查进化
            const canEvolve = student.pet.xp >= student.pet.maxXp && student.pet.level < 10;
            const evolveBtn = document.getElementById('btnEvolve');
            if (canEvolve) {
                evolveBtn.classList.remove('hidden');
                evolveBtn.disabled = false;
                evolveBtn.classList.remove('opacity-50');
            } else {
                evolveBtn.classList.add('hidden');
            }
            
            // 更新积分显示
            document.getElementById('myPoints').textContent = student.points;
            
            // 特效
            if (student.id !== GameState.currentUser.id) {
                createParticles(event.clientX, event.clientY, '❤️');
            }
        }

        // ==================== 互动操作 ====================

        function interact(type) {
            if (!selectedStudent) return;
            
            const student = selectedStudent;
            let cost = 0;
            let gain = 0;
            let msg = '';
            let emoji = '';
            
            switch(type) {
                case 'feed':
                    cost = GameState.rules.feedCost;
                    if (student.points < cost) {
                        showFloatingText('积分不足!', window.innerWidth/2, window.innerHeight/2, 'text-red-500');
                        return;
                    }
                    student.points -= cost;
                    student.pet.stats.hunger = Math.min(100, student.pet.stats.hunger + 30);
                    student.pet.stats.mood = Math.min(100, student.pet.stats.mood + 10);
                    msg = `-${cost} 喂食成功`;
                    emoji = '🍎';
                    break;
                    
                case 'play':
                    gain = GameState.rules.playGain;
                    student.points += gain;
                    student.pet.stats.mood = Math.min(100, student.pet.stats.mood + 20);
                    student.pet.stats.energy = Math.max(0, student.pet.stats.energy - 10);
                    student.pet.xp += 5;
                    msg = `+${gain} 玩耍愉快`;
                    emoji = '🎾';
                    break;
                    
                case 'study':
                    gain = GameState.rules.studyGain;
                    student.points += gain;
                    student.pet.stats.knowledge = Math.min(100, student.pet.stats.knowledge + 15);
                    student.pet.stats.energy = Math.max(0, student.pet.stats.energy - 15);
                    student.pet.xp += 15;
                    msg = `+${gain} 学习进步`;
                    emoji = '📚';
                    break;
                    
                case 'evolve':
                    if (student.pet.xp < student.pet.maxXp) return;
                    performEvolution(student);
                    return;
            }
            
            // 检查升级
            if (student.pet.xp >= student.pet.maxXp) {
                student.pet.xp = student.pet.maxXp;
            }
            
            // 更新UI
            renderPetGrid();
            selectStudent(student);
            updateUI();
            
            // 特效
            showFloatingText(msg, window.innerWidth/2, window.innerHeight - 100, 'text-2xl text-purple-600 font-black');
            createParticles(window.innerWidth/2, window.innerHeight - 150, emoji);
            
            // 震动效果
            document.body.classList.add('shake');
            setTimeout(() => document.body.classList.remove('shake'), 500);
        }

        function performEvolution(student) {
            // 显示进化动画
            const overlay = document.getElementById('evolutionOverlay');
            overlay.classList.remove('hidden');
            
            // 进度条动画
            setTimeout(() => {
                document.getElementById('evolutionProgress').style.width = '100%';
            }, 100);
            
            setTimeout(() => {
                // 执行进化
                student.pet.level++;
                student.pet.xp = 0;
                student.pet.maxXp = Math.floor(student.pet.maxXp * 1.5);
                student.pet.stage = Math.min(Math.floor((student.pet.level - 1) / 2), 3);
                
                // 解锁新形态名称
                const newStage = student.pet.type.stages[student.pet.stage];
                
                overlay.classList.add('hidden');
                document.getElementById('evolutionProgress').style.width = '0%';
                
                // 显示结果
                showEvolutionResult(student, newStage);
                
                renderPetGrid();
                selectStudent(student);
            }, 2000);
        }

        function renderShopItems(category) {
            const container = document.getElementById('shopContent');
            container.innerHTML = '';
            
            const items = SHOP_ITEMS.filter(item => item.type === category);
            
            items.forEach(item => {
                const canAfford = selectedStudent && selectedStudent.points >= item.price;
                const canUnlock = selectedStudent && selectedStudent.pet.level >= (item.unlockLevel || 1);
                const owned = selectedStudent && (
                    (item.type === 'skin' && selectedStudent.pet.skins.includes(item.id)) ||
                    (item.type === 'effect' && selectedStudent.pet.effects.includes(item.id))
                );
                
                const div = document.createElement('div');
                div.className = `shop-item ${!canAfford || !canUnlock || owned ? 'locked' : ''}`;
                
                div.innerHTML = `
                    <div class="text-4xl mb-2">${item.name.split(' ')[0]}</div>
                    <div class="font-bold text-gray-800 mb-1">${item.name.split(' ')[1]}</div>
                    <div class="text-xs text-gray-500 mb-2">${item.type === 'skin' ? '外观装扮' : item.type === 'item' ? '消耗道具' : '特殊技能'}</div>
                    <div class="flex justify-between items-center">
                        <span class="text-yellow-600 font-black">⭐ ${item.price}</span>
                        ${owned ? '<span class="text-green-500 text-xs font-bold">已拥有</span>' : 
                          !canUnlock ? `<span class="text-gray-400 text-xs">Lv.${item.unlockLevel}解锁</span>` :
                          !canAfford ? '<span class="text-red-400 text-xs">积分不足</span>' : ''}
                    </div>
                `;
                
                if (canAfford && canUnlock && !owned) {
                    div.onclick = () => buyItem(item);
                }
                
                container.appendChild(div);
            });
        }

        function buyItem(item) {
            if (!selectedStudent) return;
            
            selectedStudent.points -= item.price;
            
            if (item.type === 'skin') {
                selectedStudent.pet.skins.push(item.id);
            } else if (item.type === 'effect') {
                selectedStudent.pet.effects.push(item.id);
            }
            
            showFloatingText('购买成功!', window.innerWidth/2, window.innerHeight/2, 'text-green-500 text-2xl');
            document.getElementById('shopPoints').textContent = selectedStudent.points;
            renderShopItems(item.type + 's');
            renderPetGrid();
        }

                function renderRanking(type) {
            const container = document.getElementById('rankingContent');
            container.innerHTML = '';
            
            let sorted = [];
            if (type === 'level') {
                sorted = [...GameState.students].sort((a, b) => b.pet.level - a.pet.level || b.pet.xp - a.pet.xp);
            } else if (type === 'points') {
                sorted = [...GameState.students].sort((a, b) => b.points - a.points);
            } else if (type === 'team') {
                renderTeamRanking(container);
                return;
            }
            
            sorted.forEach((student, index) => {
                const rankClass = index === 0 ? 'rank-1' : index === 1 ? 'rank-2' : index === 2 ? 'rank-3' : '';
                const medal = index === 0 ? '🥇' : index === 1 ? '🥈' : index === 2 ? '🥉' : `<span class="text-gray-400 font-bold">${index + 1}</span>`;
                
                const div = document.createElement('div');
                div.className = `rank-item ${rankClass}`;
                div.innerHTML = `
                    <div class="text-2xl">${medal}</div>
                    <div class="w-10 h-10 rounded-full bg-gradient-to-br from-purple-400 to-pink-400 flex items-center justify-center text-lg">
                        ${student.pet.type.emoji}
                    </div>
                    <div class="flex-1">
                        <div class="font-bold text-gray-800">${student.name} <span class="text-xs text-gray-500">${student.pet.name}</span></div>
                        <div class="text-xs text-gray-500">${type === 'level' ? '等级' : '积分'}排名</div>
                    </div>
                    <div class="text-right">
                        <div class="font-black text-lg ${type === 'level' ? 'text-purple-600' : 'text-yellow-600'}">
                            ${type === 'level' ? 'Lv.' + student.pet.level : student.points}
                        </div>
                        <div class="text-xs text-gray-400">${student.teamId.replace('team', '第')}组</div>
                    </div>
                `;
                container.appendChild(div);
            });
        }

        function teacherAction(action, value) {
            switch(action) {
                case 'bonus_all':
                    GameState.students.forEach(s => {
                        s.points += value;
                        s.pet.xp += 10;
                        if (s.pet.xp > s.pet.maxXp) s.pet.xp = s.pet.maxXp;
                    });
                    showFloatingText(`全班 +${value} 分!`, window.innerWidth/2, window.innerHeight/2, 'text-3xl text-green-500 font-black');
                    createParticles(window.innerWidth/2, window.innerHeight/2, '🎉');
                    break;
                    
                case 'random_event':
                    const events = [
                        { name: '随堂测验', effect: 'xp', value: 20 },
                        { name: '大扫除', effect: 'points', value: 15 },
                        { name: '运动会', effect: 'mood', value: 30 }
                    ];
                    const event = events[Math.floor(Math.random() * events.length)];
                    
                    GameState.students.forEach(s => {
                        if (event.effect === 'xp') s.pet.xp += event.value;
                        else if (event.effect === 'points') s.points += event.value;
                        else if (event.effect === 'mood') s.pet.stats.mood = Math.min(100, s.pet.stats.mood + event.value);
                    });
                    
                    showFloatingText(`🎲 随机事件: ${event.name}! 全班+${event.value}`, window.innerWidth/2, window.innerHeight/2, 'text-2xl text-purple-600');
                    break;
                    
                case 'check_in':
                    showFloatingText('📋 签到成功! 出勤率 100%', window.innerWidth/2, window.innerHeight/2, 'text-2xl text-blue-500');
                    break;
            }
            
            updateUI();
            renderTeacherView();
        }

        function quickEdit(studentId, type, amount) {
            const student = GameState.students.find(s => s.id === studentId);
            if (!student) return;
            
            if (type === 'add') {
                student.points += amount;
                student.pet.xp += 5;
                showFloatingText(`+${amount}`, window.innerWidth/2, window.innerHeight/2, 'text-green-500');
            } else {
                student.points = Math.max(0, student.points - amount);
                student.pet.stats.mood = Math.max(0, student.pet.stats.mood - 5);
                showFloatingText(`-${amount}`, window.innerWidth/2, window.innerHeight/2, 'text-red-500');
            }
            
            renderTeacherView();
        }

        function checkMedalCondition(student, medal) {
            switch(medal.id) {
                case 'first_evolution': return student.pet.level >= 2;
                case 'rich': return student.points >= 1000;
                case 'collector': return student.pet.skins.length >= 5;
                case 'helper': return false; // 需要实现帮助系统
                case 'studious': return student.pet.stats.knowledge >= 50;
                case 'social': return false; // 需要实现互动统计
                default: return false;
            }
        }

        function updateUI() {
            // 更新顶部统计
            document.getElementById('onlineCount').textContent = GameState.students.length;
            document.getElementById('classTotalPoints').textContent = GameState.students.reduce((a,b) => a + b.points, 0).toLocaleString();

        function gameLoop() {
            // 自然衰减
            GameState.students.forEach(s => {
                // 饱食度下降影响心情
                if (Math.random() < 0.3) {
                    s.pet.stats.hunger = Math.max(0, s.pet.stats.hunger - 1);
                    if (s.pet.stats.hunger < 30) {
                        s.pet.stats.mood = Math.max(0, s.pet.stats.mood - 2);
                    }
                }
                
                // 精力自然恢复
                if (s.pet.stats.energy < 100 && Math.random() < 0.1) {
                    s.pet.stats.energy = Math.min(100, s.pet.stats.energy + 1);
                }
                
                // 知识缓慢遗忘
                if (Math.random() < 0.05) {
                    s.pet.stats.knowledge = Math.max(0, s.pet.stats.knowledge - 0.5);
                }
            });
            
            // 更新小组总分
            GameState.teams.forEach(team => {
                team.totalPoints = team.members.reduce((a, b) => a + b.points, 0);
            });
            
            // 如果当前是选中状态，更新显示
            if (selectedStudent && GameState.mode === 'individual') {
                selectStudent(selectedStudent);
            }
            
            // 随机检查勋章解锁
            checkNewMedals();
        }

        function checkNewMedals() {
            const myStudent = GameState.students.find(s => s.id === GameState.currentUser.id);
            if (!myStudent) return;
            
            MEDALS.forEach(medal => {
                if (!myStudent.medals.includes(medal.id) && checkMedalCondition(myStudent, medal)) {
                    myStudent.medals.push(medal.id);
                    showFloatingText(`🎖️ 解锁勋章: ${medal.name.split(' ')[1]}!`, window.innerWidth/2, window.innerHeight/2, 'text-2xl text-yellow-500 font-black');
                    createParticles(window.innerWidth/2, window.innerHeight/2, '🎖️');
                }
            });
        }

        }

        function showFloatingText(text, x, y, className = 'text-2xl text-purple-600 font-black') {
            const el = document.createElement('div');
            el.className = `floating-text ${className} float-up`;
            el.textContent = text;
            el.style.left = x + 'px';
            el.style.top = y + 'px';
            el.style.transform = 'translate(-50%, -50%)';
            el.style.zIndex = '9999';
            document.getElementById('effects-layer').appendChild(el);
            setTimeout(() => el.remove(), 1500);
        }

        function createParticles(x, y, emoji) {
            for (let i = 0; i < 8; i++) {
                setTimeout(() => {
                    const el = document.createElement('div');
                    el.className = 'particle emoji-float';
                    el.textContent = emoji;
                    el.style.left = (x + (Math.random() - 0.5) * 100) + 'px';
                    el.style.top = (y + (Math.random() - 0.5) * 50) + 'px';
                    el.style.setProperty('--tx', (Math.random() - 0.5) * 200 + 'px');
                    document.getElementById('effects-layer').appendChild(el);
                    setTimeout(() => el.remove(), 2000);
                }, i * 50);
            }
        }

                e.preventDefault();
